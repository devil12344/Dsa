SET SERVEROUTPUT ON;
DECLARE
ZERO_SALARY EXCEPTION;
I_EMP_ID INTEGER;
I_DOJ DATE;
I_SALARY INTEGER;
I_EXP INTEGER;
BEGIN
    I_EMP_ID := &I_EMP_ID;
    SELECT DOJ,SALARY INTO I_DOJ,I_SALARY FROM EMPLOYEE2
    WHERE EMP_ID = I_EMP_ID;
    I_EXP := TRUNC(I_DOJ - SYSDATE);
   
    IF I_SALARY < 0 THEN
        RAISE ZERO_SALARY;
    END IF;
   
    IF I_EXP >= 10 THEN
        I_SALARY := I_SALARY +(I_SALARY * 20/100);
    ELSIF I_EXP >= 5 THEN
        I_SALARY := I_SALARY + (I_SALARY * 10/100);
    ELSE
        I_SALARY := I_SALARY + (I_SALARY * 5/100);
    END IF;
   
    INSERT INTO Salary_Increment VALUES (I_EMP_ID,I_SALARY);
   
    UPDATE EMPLOYEE2 SET SALARY = I_SALARY WHERE emp_id = I_EMP_ID;
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        dbms_output.put_line('ENTER VALID ID');
    WHEN ZERO_SALARY THEN
        dbms_output.put_line('SALARY IS NOT VALID');
END;
