SET SERVEROUTPUT ON;
DECLARE
CURSOR C1 IS SELECT EMP_ID,EMP_NAME,SALARY FROM Employee WHERE SALARY > 5000;
I_EMP_ID INTEGER;
I_EMP_NAME VARCHAR(20);
I_SALARY INTEGER;
BEGIN
    OPEN C1;
    LOOP
        FETCH C1 INTO I_EMP_ID,I_EMP_NAME,I_SALARY;
        EXIT WHEN C1%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(I_EMP_ID);
        DBMS_OUTPUT.PUT_LINE(I_EMP_NAME);
        DBMS_OUTPUT.PUT_LINE(I_SALARY);
    END LOOP;
    CLOSE C1;
END;

SET SERVEROUTPUT ON;

I_ID INTEGER;
I_TOTAL_TUPLES INTEGER;

SET SERVEROUTPUT ON;
DECLARE
CURSOR C1(I_EMP_ID INTEGER) IS SELECT EMP_NAME,SALARY FROM Employee WHERE EMP_ID = I_EMP_ID;
I_EMP_ID INTEGER;
I_EMP_NAME VARCHAR(20);
I_SALARY INTEGER;
BEGIN
    I_EMP_ID := &I_EMP_ID;
    OPEN C1(I_EMP_ID);
    LOOP
        FETCH C1 INTO I_EMP_NAME,I_SALARY;
        EXIT WHEN C1%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE(I_EMP_NAME);
        DBMS_OUTPUT.PUT_LINE(I_SALARY);
    END LOOP;
    CLOSE C1;
END;
